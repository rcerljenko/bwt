cmake_minimum_required(VERSION 3.1)

project(pbwt C)
set(CMAKE_BUILD_TYPE Release)

if("${CMAKE_C_COMPILER_ID}" MATCHES "GNU")
	set(CMAKE_C_FLAGS_RELEASE "${CMAKE_C_FLAGS_RELEASE} -Wall -Wextra -s -march=native -Ofast -flto")
elseif("${CMAKE_C_COMPILER_ID}" MATCHES "Clang")
	set(CMAKE_C_FLAGS_RELEASE "${CMAKE_C_FLAGS_RELEASE} -Weverything -O4 -Wl,-s")
elseif("${CMAKE_C_COMPILER_ID}" MATCHES "Intel")
	set(CMAKE_C_FLAGS_RELEASE "${CMAKE_C_FLAGS_RELEASE} -W5 -O3 -ipo")
elseif("${CMAKE_C_COMPILER_ID}" MATCHES "MSVC")
	set(CMAKE_C_FLAGS_RELEASE "${CMAKE_C_FLAGS_RELEASE} /Wall /Ox /GF /Gd /GL /Gw /Qpar /Tc")
endif()

link_directories(libbwt/build)
add_executable(pbwt pbwt.c)
set_property(TARGET pbwt PROPERTY C_STANDARD 11)

if(UNIX OR APPLE)
	target_link_libraries(pbwt bwt pthread)
	install(TARGETS pbwt RUNTIME DESTINATION bin)
else()
	target_link_libraries(pbwt bwt)
endif()
